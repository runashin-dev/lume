    document.getElementById('r-skin-tone').textContent = d.skin_tone || '—';

    document.getElementById('r-skin-desc').textContent = d.skin_desc || '';

    if (d.eye_type) {

      document.getElementById('feat-eyelid-title').textContent = d.eye_type;

    }


    renderRatio();

    renderWidth();

    initScrollAnimation(); // 애니메이션 작동 시작


  } catch(e) {

    clearInterval(lt);

    document.getElementById('loading1').style.display = 'none';

    document.getElementById('screen-upload').classList.add('active');

    showErr(e.message);

  }

}


function classify(v, avg, tol) {

  return v - avg > tol ? 'above' : v - avg < -tol ? 'below' : 'avg';

}


function renderRatio() {

  const r = myData.ratio;

  const avgs = avgData[gender].ratio;

  const gLabel = gender === 'female' ? '여성' : '남성';


  document.getElementById('seg-top').style.flex = r.top;

  document.getElementById('seg-mid').style.flex = r.mid;

  document.getElementById('seg-bot').style.flex = r.bot;


  const items = [

    { key:'top', name:'상안부 (이마)', avg:avgs.top, color:'#7F77DD' },

    { key:'mid', name:'중안부 (눈~코)', avg:avgs.mid, color: 'var(--color-point)' },

    { key:'bot', name:'하안부 (코~턱)', avg:avgs.bot, color:'#1D9E75' }

  ];


  document.getElementById('ratio-labels').innerHTML = items.map(l => {

    const myV = r[l.key];

    const cls = classify(myV, l.avg, 1.5);

    const diffStr = cls==='above' ? `<span style="color:#639922; font-weight:600;">▲ 평균+${(myV-l.avg).toFixed(1)}%</span>`

                  : cls==='below' ? `<span style="color:#E24B4A; font-weight:600;">▼ 평균${(myV-l.avg).toFixed(1)}%</span>`

                  : `<span style="color:#888; font-weight:600;">≈ 평균 수준</span>`;

    return `<div class="ratio-label-item"><div class="ratio-dot" style="background:${l.color};"></div><div><div class="ratio-main">${l.name} — ${myV}%</div><div class="ratio-sub">${gLabel} 평균 ${l.avg}% · ${diffStr}</div></div></div>`;

  }).join('');


  document.getElementById('avg-lbl1').textContent = gLabel + ' 평균';


  document.getElementById('bars-ratio').innerHTML = items.map(l => {

    const myV = r[l.key];

    const cls = classify(myV, l.avg, 1.5);

    const scale = v => Math.max(0, Math.min(100, ((v-20)/30*100))).toFixed(1);

    const myW = scale(myV), avgW = scale(l.avg), gldW = scale(33.3);

    const fillCls = cls==='above'?'fill-above':cls==='below'?'fill-below':'fill-avg';

    const valCls  = cls==='above'?'val-above' :cls==='below'?'val-below' :'val-avg';

    const status  = cls==='above'?'평균보다 높음':cls==='below'?'평균보다 낮음':'평균 수준';

    return `<div class="bar-section">

      <div class="bar-label-row"><span class="bar-name">${l.name} 비율</span><span class="bar-value ${valCls}">${myV}%</span></div>

      <div class="bar-wrap">

        <div class="bar-fill ${fillCls}" data-w="${myW}" style="width:0%;"></div>

        <div class="avg-mark" style="left:${avgW}%;"><div class="avg-mark-line"></div><div class="avg-mark-label">${gLabel} ${l.avg}%</div></div>

        <div class="golden-mark" style="left:${gldW}%;"><div class="golden-mark-line"></div><div class="golden-mark-label">황금 33.3%</div></div>

      </div>

      <div class="bar-desc">${status} · 내 수치 ${myV}% (기준선과 비교해 보세요)</div>

    </div>`;

  }).join('');


  setTimeout(() => animateBars('bars-ratio'), 200);

}


function renderWidth() {

  const avgs = avgData[gender].width;

  const gLabel = gender === 'female' ? '여성' : '남성';

  document.getElementById('avg-lbl2').textContent = gLabel + ' 평균';


  document.getElementById('bars-width').innerHTML = Object.entries(myData.width).map(([name, my]) => {

    const avg = avgs[name];

    const cls = classify(my.v, avg, 4);

    const fillCls = cls==='above'?'fill-above':cls==='below'?'fill-below':'fill-avg';

    const valCls  = cls==='above'?'val-above' :cls==='below'?'val-below' :'val-avg';

    const status  = cls==='above'?'평균보다 넓음':cls==='below'?'평균보다 좁음':'평균 수준';

    return `<div class="bar-section">

      <div class="bar-label-row"><span class="bar-name">${name}</span><span class="bar-value ${valCls}">${my.l}</span></div>

      <div class="bar-wrap">

        <div class="bar-fill ${fillCls}" data-w="${my.v}" style="width:0%;"></div>

        <div class="avg-mark" style="left:${avg}%;"><div class="avg-mark-line"></div><div class="avg-mark-label">${gLabel} 평균</div></div>

      </div>

      <div class="bar-desc">${status} · 내 수치 ${my.v} vs ${gLabel} 평균 ${avg}</div>

    </div>`;

  }).join('');


  setTimeout(() => animateBars('bars-width'), 200);

}


function animateBars(id) {

  document.getElementById(id).querySelectorAll('[data-w]').forEach(el => el.style.width = el.dataset.w + '%');

}


function switchTab(id, el) {

  document.querySelectorAll('.rtab').forEach(t => t.classList.remove('active'));

  document.getElementById('tab-' + id).classList.add('active');

  el.classList.add('active');

  

  // 탭 전환 시 새로 노출된 카드들의 애니메이션 재확인

  initScrollAnimation();

}
